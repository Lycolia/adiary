//////////////////////////////////////////////////////////////////////////////
// ●CSSから値を取得する
//////////////////////////////////////////////////////////////////////////////
function get_value_from_css(id, attr) {
	var span = $('<span>').attr('id', id).css('display', 'none');
	$('#body').append(span);
	if (attr) {
		attr = span.css(attr);
		span.remove();
		return attr;
	}
	var size = span.css('font-size');	// 1pxの時のみ有効
	var str  = span.css('font-family');
	span.remove();
	if (str == null || size != '1px') return '';
	str = str.replace(/["']/g, '');
	return str;
}

拡張リスト記法

通常のリスト記法では、リスト中に改行を入れたりブロックを展開することができませんでした。拡張リスト記法では、それらが可能になります。
  • 変数x
    xは未知変数である
  • 変数y
    yはxについての２次関数。
    y=ax2+bx+c
定義済リストも同じように拡張記法を使えます。
